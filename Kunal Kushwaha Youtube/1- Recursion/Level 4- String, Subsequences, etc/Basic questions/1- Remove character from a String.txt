package com.kamble;

/*
    Remove character from a String.
 */

public class Main {
    public static void main(String[] args) {

        String str = "baccad";

        System.out.println(func(str, 'a'));
    }

    private static String func(String str, char x) {

        StringBuilder sb = new StringBuilder(str);
         sb = _func(sb, x, str.length()-1);

         return sb.toString();
    }

    private static StringBuilder _func(StringBuilder str, char x, int n) {

        if(n==-1) return str;

        //check whether the char at current index is the one.
        if(x == str.charAt(n)) {
            //remove the character
            str.deleteCharAt(n);
        }

        return _func(str, x, n-1);
    }
}
