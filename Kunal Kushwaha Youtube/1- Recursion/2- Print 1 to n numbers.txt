package com.kamble.demo;

/*
    Problem
        Print 1 to n numbers
 */

public class Main {
    public static void main(String[] args) {

        print(5);
    }

    private static void print(int n) {

        if(n==0) return;

        print(n-1); //first print the previous number
        System.out.println(n);
    }
}
