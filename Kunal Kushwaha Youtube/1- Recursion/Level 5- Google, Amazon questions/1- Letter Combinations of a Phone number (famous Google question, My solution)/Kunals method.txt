import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

public class Main {
    public static void main(String[] args) {
        phoneNumberLetters("", "23");
    }

    private static void phoneNumberLetters(String ans, String given) {

        Map<Integer, String> map = new HashMap();
        map.put(2, "abc");
        map.put(3, "def");
        map.put(4, "ghi");
        map.put(5, "jkl");
        map.put(6, "mno");
        map.put(7, "pqrs");
        map.put(8, "tuv");
        map.put(9, "wxyz");

        funcPrint(ans, given, map);
        System.out.println(funcList(ans, given, map));
    }

    private static void funcPrint(String ans, String given, Map<Integer, String> map) {

        if(given.isEmpty()) {
            System.out.println(ans);
            return;
        }

        int digit = Integer.parseInt(Character.toString(given.charAt(0)));
        String lettersForThatDigit = map.get(digit);

        //If 1 has 3 letters make 3 functions calls
        for (int i = 0; i <lettersForThatDigit.length() ; i++) {
            funcPrint(ans + lettersForThatDigit.charAt(i), given.substring(1), map);
        }
    }


    private static List<String> funcList(String ans, String given, Map<Integer, String> map) {

        List<String> list = new ArrayList<>();
        if(given.isEmpty()) {
            list.add(ans);
            return list;
        }

        int digit = Integer.parseInt(Character.toString(given.charAt(0)));
        String lettersForThatDigit = map.get(digit);

        //If 1 has 3 letters make 3 functions calls
        for (int i = 0; i <lettersForThatDigit.length() ; i++) {
            List<String> subList = funcList(ans + lettersForThatDigit.charAt(i), given.substring(1), map);
            list.addAll(subList);
        }

        return list;
    }
}
