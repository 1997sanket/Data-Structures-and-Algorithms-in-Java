 public void insertAtIndexRec(int data, int index) {

        _insertAtIndexRec(head, data, index, 0);
    }

    private void _insertAtIndexRec(Node node, int data, int index, int i) {

        if(index == 0) {
            insertFirst(data);
            return;
        }

        if(index == size-1) {
            insertLast(data);
            return;
        }

        if(i == (index-1)) {
            Node n = new Node(data);
            n.next = node.next;
            node.next = n;
            size++;

            return;
        }

        _insertAtIndexRec(node.next, data, index, i+1);
    }
