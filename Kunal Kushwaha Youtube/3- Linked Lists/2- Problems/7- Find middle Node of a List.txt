 public Node findMiddle() {

        boolean flag = false;

        Node slow = head;
        Node fast = head;

        while(fast!=null && fast.next != null) {

            if(slow == fast && slow!=head) {
                flag = true;
                break;
            }

            slow = slow.next;
            fast = fast.next.next;
        }

        return slow;
    }
