 public LL.Node findWhereTheCycleBegins() {

        //first find the length of the cycle
        int l = findLengthOfCycle();

        if(l==0) return null;

        //create two pointers are head
        Node p1 = head;
        Node p2 = head;

        //traverse the firt pointer till "l" steps
        while(l!=0) {
            p1 = p1.next;
            l--;
        }

        //now traverse both till they meet
        while(p1!=p2) {
            p1 = p1.next;
            p2 = p2.next;
        }

        //return any 1 of the pointers (the position where they now meet, will be beginning of the cycle)
        return p1;
    }
